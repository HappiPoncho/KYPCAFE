120      private JPanel makeRow(String name, String qty, String price, boolean bold) {
121          JPanel row = new JPanel(new BorderLayout());
122          row.setBackground(Color.WHITE);
123          row.setMaximumSize(new Dimension(Integer.MAX_VALUE, 20));
124  
125          int style = bold ? Font.BOLD : Font.PLAIN;
126  
127          JLabel nameLabel  = new JLabel(name);
128          JLabel qtyLabel   = new JLabel(qty, JLabel.CENTER);
129          JLabel priceLabel = new JLabel(price, JLabel.RIGHT);
130  
131          nameLabel.setFont(new Font("Arial", style, 12));
132          qtyLabel.setFont(new Font("Arial", style, 12));
133          priceLabel.setFont(new Font("Arial", style, 12));
134  
135          qtyLabel.setPreferredSize(new Dimension(40, 20));
136          priceLabel.setPreferredSize(new Dimension(70, 20));
137  
138          JPanel right = new JPanel(new BorderLayout());
139          right.setBackground(Color.WHITE);
140          right.add(qtyLabel, BorderLayout.WEST);
141          right.add(priceLabel, BorderLayout.EAST);
142          right.setPreferredSize(new Dimension(110, 20));
143  
144          row.add(nameLabel, BorderLayout.CENTER);
145          row.add(right, BorderLayout.EAST);
146          return row;
147      }

156      private JPanel buildButtonPanel() {
157          JPanel panel = new JPanel(new GridLayout(1, 2, 10, 0));
158          panel.setBackground(Color.WHITE);
159          panel.setBorder(new EmptyBorder(0, 24, 20, 24));
160  
161          JButton closeBtn = new JButton("Close");
162          closeBtn.setFocusPainted(false);
163          closeBtn.addActionListener(e -> {
164              dispose();
165              parent.dispose();
166              SwingUtilities.invokeLater(() -> new WelcomePanel());
167          });
168  
169          JButton downloadBtn = new JButton("Download PDF");
170          downloadBtn.setBackground(new Color(90, 60, 30));
171          downloadBtn.setForeground(Color.WHITE);
172          downloadBtn.setFocusPainted(false);
173          downloadBtn.setBorderPainted(false);
174          downloadBtn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
175          downloadBtn.addActionListener(e -> savePDF());
176  
177          panel.add(closeBtn);
178          panel.add(downloadBtn);
179          return panel;
180      }

182      private void savePDF() {
183          JFileChooser chooser = new JFileChooser();
184          chooser.setSelectedFile(new File(dtime +"KPYReciept.pdf"));
185          int result = chooser.showSaveDialog(this);
186          if (result != JFileChooser.APPROVE_OPTION) return;
187  
188          File file = chooser.getSelectedFile();
189          if (!file.getName().endsWith(".pdf")) file = new File(file.getAbsolutePath() + ".pdf");
190  
191          try {
192              Document doc = new Document(PageSize.A6);
193              PdfWriter.getInstance(doc, new FileOutputStream(file));
194              doc.open();
195  
196              com.itextpdf.text.Font titleFont  = new com.itextpdf.text.Font(com.itextpdf.text.Font.FontFamily.HELVETICA, 18, com.itextpdf.text.Font.BOLD);
197              com.itextpdf.text.Font subFont    = new com.itextpdf.text.Font(com.itextpdf.text.Font.FontFamily.HELVETICA, 10, com.itextpdf.text.Font.NORMAL, BaseColor.GRAY);
198              com.itextpdf.text.Font headerFont = new com.itextpdf.text.Font(com.itextpdf.text.Font.FontFamily.HELVETICA, 10, com.itextpdf.text.Font.BOLD);
199              com.itextpdf.text.Font bodyFont   = new com.itextpdf.text.Font(com.itextpdf.text.Font.FontFamily.HELVETICA, 10);
200              com.itextpdf.text.Font totalFont  = new com.itextpdf.text.Font(com.itextpdf.text.Font.FontFamily.HELVETICA, 12, com.itextpdf.text.Font.BOLD);
201  
202              Paragraph shopName = new Paragraph("KiPeYe Coffee", titleFont);
203              shopName.setAlignment(Element.ALIGN_CENTER);
204              doc.add(shopName);
205  
206              Paragraph receipt = new Paragraph("Official Receipt", subFont);
207              receipt.setAlignment(Element.ALIGN_CENTER);
208              doc.add(receipt);
209  
210              Paragraph dt = new Paragraph(dateTime, subFont);
211              dt.setAlignment(Element.ALIGN_CENTER);
212              dt.setSpacingAfter(6);
213              doc.add(dt);
214  
215              Paragraph type = new Paragraph(orderType.toUpperCase(), headerFont);
216              type.setAlignment(Element.ALIGN_CENTER);
217              type.setSpacingAfter(10);
218              doc.add(type);
219  
220              PdfPTable table = new PdfPTable(3);
221              table.setWidthPercentage(100);
222              table.setWidths(new float[]{5, 1.5f, 2});
223              table.setSpacingBefore(6);
224              table.setSpacingAfter(6);
225  
226              for (String h : new String[]{"Item", "Qty", "Price"}) {
227                  PdfPCell cell = new PdfPCell(new Phrase(h, headerFont));
228                  cell.setBorder(Rectangle.BOTTOM);
229                  cell.setPadding(4);
230                  table.addCell(cell);
231              }
232  
233              for (Orders item : items) {
234                  PdfPCell c1 = new PdfPCell(new Phrase(item.getMenu().getName(), bodyFont));
235                  PdfPCell c2 = new PdfPCell(new Phrase("x" + item.getQuantity(), bodyFont));
236                  PdfPCell c3 = new PdfPCell(new Phrase("₱" + item.getTotalPrice(), bodyFont));
237                  c2.setHorizontalAlignment(Element.ALIGN_CENTER);
238                  c3.setHorizontalAlignment(Element.ALIGN_RIGHT);
239                  for (PdfPCell c : new PdfPCell[]{c1, c2, c3}) {
240                      c.setBorder(Rectangle.NO_BORDER);
241                      c.setPadding(3);
242                  }
243                  table.addCell(c1);
244                  table.addCell(c2);
245                  table.addCell(c3);
246              }
247  
248              doc.add(table);
249  
250              Paragraph divider = new Paragraph("________________________________________________", subFont);
251              divider.setAlignment(Element.ALIGN_CENTER);
252              doc.add(divider);
253  
254              Paragraph totalPara = new Paragraph("TOTAL: ₱" + total, totalFont);
255              totalPara.setAlignment(Element.ALIGN_RIGHT);
256              totalPara.setSpacingBefore(8);
257              doc.add(totalPara);
258  
259              Paragraph thanks = new Paragraph("Thank you for your order!", subFont);
260              thanks.setAlignment(Element.ALIGN_CENTER);
261              thanks.setSpacingBefore(16);
262              doc.add(thanks);
263  
264              doc.close();
265              JOptionPane.showMessageDialog(this, "Receipt saved to:\n" + file.getAbsolutePath());
266  
267          } catch (Exception ex) {
268              JOptionPane.showMessageDialog(this, "Failed to save PDF:\n" + ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
269          }
270      }
