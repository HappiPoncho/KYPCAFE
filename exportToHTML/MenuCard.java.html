1    import javax.swing.*;
2    import java.awt.*;
3    import java.awt.event.*;
4    
5    public class MenuCard extends JPanel {
6    
7        public MenuCard(Menu item, OrderManager orderManager) {
8            setLayout(new BorderLayout());
9            setPreferredSize(new Dimension(200, 200));
10           setBackground(Color.WHITE);
11           setBorder(BorderFactory.createLineBorder(new Color(220, 220, 220), 1, true));
12   
13           JLabel imageLabel = new JLabel();
14           imageLabel.setHorizontalAlignment(JLabel.CENTER);
15           ImageIcon icon = new ImageIcon(item.getImagePath());
16           Image img = icon.getImage().getScaledInstance(140, 100, Image.SCALE_SMOOTH);
17           imageLabel.setIcon(new ImageIcon(img));
18           add(imageLabel, BorderLayout.CENTER);
19   
20           JPanel textPanel = new JPanel(new GridLayout(2, 1));
21           textPanel.setBackground(Color.WHITE);
22           JLabel nameLabel = new JLabel(item.getName());
23           nameLabel.setFont(new Font("Arial", Font.BOLD, 12));
24           JLabel priceLabel = new JLabel(item.getPrice());
25           priceLabel.setForeground(Color.GRAY);
26           textPanel.add(nameLabel);
27           textPanel.add(priceLabel);
28           add(textPanel, BorderLayout.SOUTH);
29   
30           addMouseListener(new MouseAdapter() {
31               public void mouseClicked(MouseEvent e) {
32                   orderManager.addItem(item);
33               }
34               public void mouseEntered(MouseEvent e) { setBackground(new Color(230, 230, 230)); }
35               public void mouseExited(MouseEvent e) { setBackground(Color.WHITE); }
36           });
37       }
38   }
