9    public class CategoryPanel extends JPanel {

12       public CategoryPanel(Category[] categories, Consumer<Category> onCategorySelected) {
13           setLayout(new GridLayout(1, categories.length, 8, 0));
14           setBackground(new Color(239, 233, 227));
15           setBorder(BorderFactory.createEmptyBorder(0, 0, 10, 0));
16   
17       buttons = new CategoryBtn[categories.length];
18   
19           for (int i = 0; i < categories.length; i++) {
20               final int index = i;
21               buttons[i] = new CategoryBtn(categories[i], () -> {
22                   selectButton(index);
23                   onCategorySelected.accept(categories[index]);
24               });
25               add(buttons[i]);
26           }
27   
28           if (categories.length > 0) {
29               buttons[0].setSelected(true);
30           }
31       }
32   
33       private void selectButton(int index) {
34           for (CategoryBtn btn : buttons) btn.setSelected(false);
35           buttons[index].setSelected(true);
36       }
37   }
