1    import javax.swing.*;

3    import java.awt.image.BufferedImage;
4    import javax.imageio.ImageIO;

8    
9    public class WelcomePanel extends JFrame {
10   
11   
12       public WelcomePanel() {
13   
14           FlatMacLightLaf.setup();
15   
16           setTitle("KiPeYe Coffee");
17           setSize(500, 380);
18           setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
19           setLocationRelativeTo(null);
20           setResizable(false);
21           setLayout(new BorderLayout());
22   
23           JPanel bg = new JPanel(new BorderLayout()) {
24   
25               BufferedImage wallpaper;
26               BufferedImage logo;
27   
28               {
29                   try {
30                       wallpaper = ImageIO.read(new File("rsrc/icons and fonts/wallpaper.png"));
31                   } catch (IOException e) {
32                       wallpaper = null;
33                   }
34   
35                   try {
36                       logo = ImageIO.read(new File("rsrc/icons and fonts/llogo.png"));
37                   } catch (IOException e) {
38                       logo = null;
39                   }
40               }
41   
42               @Override
43               protected void paintComponent(Graphics g) {
44                   super.paintComponent(g);
45                   Graphics2D g2d = (Graphics2D) g.create();
46   
47                   if (wallpaper != null) {
48                       g2d.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 0.45f));
49                       g2d.drawImage(wallpaper, 0, 0, getWidth(), getHeight(), this);
50                   }
51   
52                   if (logo != null) {
53                       g2d.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 0.15f));
54                       int size = Math.min(getWidth(), getHeight());
55                       int x = (getWidth() - size) / 2;
56                       int y = (getHeight() - size) / 2;
57                       g2d.drawImage(logo, x, y, size, size, this);
58                   }
59   
60                   g2d.dispose();
61               }
62           };
63   
64           bg.setBackground(new Color(250, 245, 230));
65           setContentPane(bg);
66   
67           Color brown = new Color(90, 60, 30);
68           Color cream = new Color(255, 253, 208);
69   
70           JPanel centerPanel = new JPanel();
71           centerPanel.setLayout(new BoxLayout(centerPanel, BoxLayout.Y_AXIS));
72           centerPanel.setOpaque(false);
73           centerPanel.setBorder(BorderFactory.createEmptyBorder(60, 40, 20, 40));
74   
75           JLabel shopName = new JLabel("KiPeYe Coffee");
76           shopName.setFont(Fonts.of(Font.BOLD, 80f));
77           shopName.setForeground(brown);
78           shopName.setAlignmentX(Component.CENTER_ALIGNMENT);
79   
80           JLabel tagline = new JLabel("one cup can fill you up");
81           tagline.setFont(new Font("Arial", Font.ITALIC, 14));
82           tagline.setForeground(Color.BLACK);
83           tagline.setAlignmentX(Component.CENTER_ALIGNMENT);
84   
85           JLabel prompt = new JLabel("How will you be dining today?");
86           prompt.setFont(new Font("Arial", Font.PLAIN, 13));
87           prompt.setForeground(Color.BLACK);
88           prompt.setAlignmentX(Component.CENTER_ALIGNMENT);
89   
90           centerPanel.add(shopName);
91           centerPanel.add(Box.createVerticalStrut(6));
92           centerPanel.add(tagline);
93           centerPanel.add(Box.createVerticalStrut(30));
94           centerPanel.add(prompt);
95   
96           bg.add(centerPanel, BorderLayout.CENTER);
97   
98           JPanel btnPanel = new JPanel(new GridLayout(1, 2, 16, 0));
99           btnPanel.setOpaque(false);
100          btnPanel.setBorder(BorderFactory.createEmptyBorder(10, 60, 50, 60));
101  
102          JButton dineInBtn  = makeButton("Dine In",  brown, Color.WHITE);
103          JButton takeOutBtn = makeButton("Take Out", cream, brown);
104  
105          dineInBtn.addActionListener(e  -> proceed("Dine In"));
106          takeOutBtn.addActionListener(e -> proceed("Take Out"));
107  
108          btnPanel.add(dineInBtn);
109          btnPanel.add(takeOutBtn);
110  
111          bg.add(btnPanel, BorderLayout.SOUTH);
112  
113          JPanel upPanel = new JPanel(new BorderLayout());
114          upPanel.setOpaque(false);
115          upPanel.setBorder(BorderFactory.createEmptyBorder(8, 12, 8, 12));
116  
117          JButton exitBtn = makeButton("X", Color.red, Color.WHITE);
118          exitBtn.addActionListener(e -> {
119              int confirm = JOptionPane.showConfirmDialog(
120                      this,
121                      "Are you sure you want to exit?",
122                      "Exit",
123                      JOptionPane.YES_NO_OPTION
124              );
125              if (confirm == JOptionPane.YES_OPTION) System.exit(0);
126          });
127  
128          upPanel.add(exitBtn, BorderLayout.EAST);
129  
130          bg.add(upPanel, BorderLayout.NORTH);
131  
132          setVisible(true);
133      }
134  
135      private JButton makeButton(String text, Color bg, Color fg) {
136          JButton btn = new JButton(text);
137          btn.setFont(new Font("Arial", Font.BOLD, 14));
138          btn.setBackground(bg);
139          btn.setForeground(fg);
140          btn.setFocusPainted(false);
141          btn.setBorderPainted(false);
142          btn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
143          return btn;
144      }
145  
146  
147  
148      private void proceed(String orderType) {
149          dispose();
150          SwingUtilities.invokeLater(() -> new Main().createUI(orderType));
151      }
