1    import javax.swing.*;
2    import java.awt.*;
3    import java.awt.event.MouseAdapter;
4    import java.awt.event.MouseEvent;
5    
6    public class CategoryBtn extends JPanel {
7        private static final Color DEFAULT_BG = new Color(228, 218, 200);
8        private static final Color SELECTED_BG = new Color(217, 207, 199);
9        private static final Color HOVER_BG = new Color(202, 194, 178);
10   
11       private boolean selected = false;
12   
13       public CategoryBtn(Category category, Runnable onSelect) {
14           setLayout(new BorderLayout(0, 4));
15           setBackground(DEFAULT_BG);
16           setBorder(BorderFactory.createCompoundBorder(
17                   BorderFactory.createLineBorder(new Color(200, 200, 200), 1, true),
18                   BorderFactory.createEmptyBorder(8, 6, 8, 6)
19           ));
20           setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
21   
22           // IMAGE
23           JLabel imageLabel = new JLabel();
24           imageLabel.setHorizontalAlignment(JLabel.CENTER);
25           ImageIcon icon = new ImageIcon(category.getImagePath());
26           Image img = icon.getImage().getScaledInstance(48, 48, Image.SCALE_SMOOTH);
27           imageLabel.setIcon(new ImageIcon(img));
28   
29           // LABEL
30           JLabel nameLabel = new JLabel(category.getName(), JLabel.CENTER);
31           nameLabel.setFont(new Font("Arial", Font.BOLD, 11));
32   
33           add(imageLabel, BorderLayout.CENTER);
34           add(nameLabel, BorderLayout.SOUTH);
35   
36           addMouseListener(new MouseAdapter() {
37               public void mouseClicked(MouseEvent e) { onSelect.run(); }
38               public void mouseEntered(MouseEvent e) { if (!selected) setBackground(HOVER_BG); }
39               public void mouseExited(MouseEvent e)  { if (!selected) setBackground(DEFAULT_BG); }
40           });
41       }
42   
43       public void setSelected(boolean selected) {
44           this.selected = selected;
45           setBackground(selected ? SELECTED_BG : DEFAULT_BG);
46       }
47   }
