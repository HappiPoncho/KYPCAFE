1    import javax.swing.*;
2    import java.awt.*;
3    import java.util.ArrayList;
4    import java.util.List;
5    
6    public class OrderPanel extends JPanel {
7        private OrderManager orderManager;
8        private JPanel itemsPanel;
9        private JLabel totalLabel;
10       private String orderType;
11   
12       private static final Color CREAM = new Color(255, 253, 208);
13   
14       public OrderPanel(OrderManager orderManager,String orderType) {
15           this.orderManager = orderManager;
16           this.orderType = orderType;
17           setLayout(new BorderLayout(0, 10));
18           setBackground(CREAM);
19           setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
20   
21           // TITLE
22           JLabel title = new JLabel("Your Order");
23           title.setFont(new Font("Arial", Font.BOLD, 16));
24           title.setBorder(BorderFactory.createEmptyBorder(0, 0, 8, 0));
25           add(title, BorderLayout.NORTH);
26   
27           // Items list
28           itemsPanel = new JPanel();
29           itemsPanel.setLayout(new BoxLayout(itemsPanel, BoxLayout.Y_AXIS));
30           itemsPanel.setBackground(CREAM);
31   
32           JScrollPane scroll = new JScrollPane(itemsPanel);
33           scroll.setBorder(null);
34           scroll.setBackground(CREAM);
35           add(scroll, BorderLayout.CENTER);
36   
37           JPanel bottomPanel = new JPanel(new BorderLayout(0, 8));
38           bottomPanel.setBackground(CREAM);
39           bottomPanel.setBorder(BorderFactory.createEmptyBorder(8, 0, 0, 0));
40   
41           totalLabel = new JLabel("Total: ₱0");
42           totalLabel.setFont(new Font("Arial", Font.BOLD, 14));
43   
44           JButton checkoutBtn = new JButton("Checkout");
45           checkoutBtn.setFont(new Font("Arial", Font.BOLD, 13));
46           checkoutBtn.setBackground(new Color(90, 60, 30));
47           checkoutBtn.setForeground(Color.WHITE);
48           checkoutBtn.setFocusPainted(false);
49           checkoutBtn.setBorderPainted(false);
50           checkoutBtn.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
51           checkoutBtn.setPreferredSize(new Dimension(0, 38));
52           checkoutBtn.addActionListener(e -> {
53               if (orderManager.getOrderItems().isEmpty()) {
54                   JOptionPane.showMessageDialog(this, "No items in order!", "Checkout", JOptionPane.WARNING_MESSAGE);
55                   return;
56               }
57               int confirm = JOptionPane.showConfirmDialog(this,
58                       "Total: ₱" + orderManager.getTotal() + "\nProceed to checkout?",
59                       "Checkout", JOptionPane.YES_NO_OPTION);
60               // replace the YES_OPTION block with this
61               if (confirm == JOptionPane.YES_OPTION) {
62                   List<Orders> snapshot = new ArrayList<>(orderManager.getOrderItems());
63                   int currentTotal = orderManager.getTotal();
64   
65                   orderManager.getOrderItems().clear();
66                   orderManager.notifyListenCheckout();
67   
68                   new Receipt((JFrame) SwingUtilities.getWindowAncestor(this), snapshot, currentTotal, orderType);
69               }
70           });
71   
72           bottomPanel.add(totalLabel, BorderLayout.NORTH);
73           bottomPanel.add(checkoutBtn, BorderLayout.SOUTH);
74           add(bottomPanel, BorderLayout.SOUTH);
75   
76           orderManager.addChangeListener(this::refresh);
77       }
78   
79       private void refresh() {
80           itemsPanel.removeAll();
81   
82           for (Orders item : orderManager.getOrderItems()) {
83               itemsPanel.add(new OrderRows(item, orderManager));
84               itemsPanel.add(Box.createVerticalStrut(5));
85           }
86   
87           totalLabel.setText("Total: ₱" + orderManager.getTotal());
88           itemsPanel.revalidate();
89           itemsPanel.repaint();
90       }
91   }
